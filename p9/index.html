  কোনো জাভা অবজেক্টেকে স্ট্রিংয়ে রূপান্তর করতে চাইলে toString() মেথড কল করা হয়। এটি নানা কাজে বিশেষ করে ডিবাগিং কাজে ব্যবহার করা হয়।

public Employee(String name, int age) {
    this.name = name;

}

} 

public Employee(String name, int age) {
    this.name = name;
    this.age = age;
}

@Override
public String toString() {

    return name + ": " + age;
}

public static void main(String[] args) {
    Employee employee = new Employee("Rahim", 27);
    System.out.println(employee.toString());
}

}
